    {

    }

        LoginPanel panel = UIManager.GetPanel<LoginPanel>();//得到面板

        panel.SetPassword("123456");//设置属性
    }

    private void GetLayer()
    {
        UIManager.GetLayer(UILayer.Layer1);//获取层级
        Transform common = UIManager.GetLayer(UILayer.Layer2);
    }

    private void Clear()
